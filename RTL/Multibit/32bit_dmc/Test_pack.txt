`timescale 1ns/1ps

module tb_memory_wrapper;

  // -------------------------------------------------
  // ---------------  Parameters  --------------------
  // -------------------------------------------------
  localparam int ADDR_WIDTH = 3;
  localparam int DATA_WIDTH = 16;
  localparam int DEPTH      = 1 << ADDR_WIDTH;

  // -------------------------------------------------
  // ------------  Testbench signals -----------------
  // -------------------------------------------------
  logic                   clk;
  logic                   cen;      // active low
  logic                   wen;      // write enable
  logic [ADDR_WIDTH-1:0]  addr;
  logic [DATA_WIDTH-1:0]  d;
  logic [DATA_WIDTH-1:0]  q;
  logic                   error_corrected;
  logic                   double_bit_error;

  // -------------------------------------------------
  // ---------------  DUT instance -------------------
  // -------------------------------------------------
  memory_wrapper #(
    .ADDR_WIDTH(ADDR_WIDTH),
    .DATA_WIDTH(DATA_WIDTH)
  ) dut (
    .clk                (clk),
    .cen                (cen),
    .wen                (wen),
    .addr               (addr),
    .d                  (d),
    .q                  (q),
    .error_corrected(error_corrected)
  );

  // -------------------------------------------------
  // ----------    Clock generation ------------------
  // -------------------------------------------------
  initial clk = 0;
  always #5 clk = ~clk; // 100 MHz clock

  // --------------------------------------------------------
  // ------------------  Task: Write to memory  -------------
  // --------------------------------------------------------
  task automatic write_mem(input logic [ADDR_WIDTH-1:0] waddr,
                           input logic [DATA_WIDTH-1:0] wdata);
    begin
      d    = wdata;
      addr = waddr;
      @(posedge clk);
      cen  = 1'b0;  // enable chip (active low)
      wen  = 1'b0;  // write enable
      @(posedge clk); // one cycle write
      cen  = 1'b1;  // disable after write

    end
  endtask
  
  // --------------------------------------------------------
  // ------------------  Task: Write to memory  -------------
  // --------------------------------------------------------
  //This task flips the bit position where theres a 1 in wdata
  task automatic generate_error(input logic [ADDR_WIDTH-1:0] waddr,
                                input logic [DATA_WIDTH-1:0] wdata);
    
    begin
        dut.u_sram.mem[waddr] = dut.u_sram.mem[waddr] ^ wdata;
    end
  endtask

  // -----------------------------------------------------------
  // -------------------  Task: Read from memory ---------------
  // -----------------------------------------------------------
  task automatic read_mem(input logic [ADDR_WIDTH-1:0] raddr,
                          output logic [DATA_WIDTH-1:0] rdata);
    begin
      @(posedge clk);
      cen  = 1'b0; // enable chip
      wen  = 1'b1; // disable write
      addr = raddr;
	  @(posedge clk);
      rdata = q;
      cen   = 1'b1;
	  // Wait some cycles
	  repeat (5) @(posedge clk); //try to match with delay*
    end
  endtask

  // -------------------------------------------------
  // ----------------- Stimulus ----------------------
  // -------------------------------------------------
  initial begin
    logic [DATA_WIDTH-1:0] read_data;

    // Initialize signals
    cen  = 1'b1;
    wen  = 1'b0;
    addr = '0;
    d    = '0;

    // Wait some cycles
    repeat (5) @(posedge clk);

    // -------------------------------------------
    // ----------  Write a value -----------------
    // -------------------------------------------
    $display("[%0t] Writing 0xDEADBEEF to address 0", $time);
    write_mem(7'h00, 32'hDEADBEEF);
	
    @(posedge clk);
    // -------------------------------------------
    // ----------  Generate error ----------------
    // -------------------------------------------
    generate_error(7'h00,32'h01 );
    $display("[%0t] Generating error 0x01 to address 0", $time);
    // -------------------------------------------
    // ----------  Read it back  -----------------
    // -------------------------------------------
    read_mem(7'h00, read_data);
    $display("[%0t] Read data = 0x%h", $time, read_data);

    if (read_data !== (32'hDEADBEEF &(1 << DATA_WIDTH) - 1) ) 
      $error("Read data mismatch!");
      
    // Wait some cycles
    repeat (5) @(posedge clk);

    // -------------------------------------------
    // ----------  Write a value -----------------
    // -------------------------------------------
    $display("[%0t] Writing 0xDEADBEEF to address 0", $time);
    write_mem(7'h01, 32'hBAADF00D);
	
    @(posedge clk);
    // -------------------------------------------
    // ----------  Generate error ----------------
    // -------------------------------------------
    generate_error(7'h01,32'h03 );
    $display("[%0t] Generating error 0x03 to address 1", $time);
    // -------------------------------------------
    // ----------  Read it back  -----------------
    // -------------------------------------------
    read_mem(7'h01, read_data);
    $display("[%0t] Read data = 0x%h", $time, read_data);

    if (read_data !== (32'hBAADF00D &(1 << DATA_WIDTH) - 1) )
      $error("Read data mismatch!");

    $finish;
  end

endmodule
